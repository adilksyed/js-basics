const ball = document.getElementById('ball'); document.addEventListener('keydown' , handleKeyPress); function handleKeyPress(e) { if (e.code == 'ArrowLeft') { position = position - 10; } if (e.code == 'ArrowRight'){ position = position + 10; } If (position < 0) { position = 0; } refresh(); } function refresh() { ball.style.left = position + 'px'; }
